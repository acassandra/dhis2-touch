import { Component, OnInit, Input } from '@angular/core';
import { CurrentEntrySelection, EntryFormStatus } from 'src/models';

@Component({
  selector: 'app-data-entry-form-summary',
  templateUrl: './data-entry-form-summary.component.html',
  styleUrls: ['./data-entry-form-summary.component.scss']
})
export class DataEntryFormSummaryComponent implements OnInit {
  @Input() currentEntrySelection: CurrentEntrySelection;
  @Input() entryFormSyncStatus: EntryFormStatus;

  isSyncStatusVisible: boolean;

  constructor() {}

  ngOnInit() {
    this.isSyncStatusVisible = this.entryFormSyncStatus ? true : false;
  }
}
